## Some nice examples

Print("  Coxeter geometry  \n");
g := PSL(2,19);
maxs := MaximalSubgroupClassReps(g);;
a5s := Filtered(maxs, t->Size(t)=60);
g1 := Group([ (2,5,20)(3,14,13)(4,9,19)(6,7,11)(8,10,17)(12,18,16), 
              (1,18)(2,6)(3,12)(4,15)(5,17)(7,19)(8,16)(9,20)(10,13)(11,14) ]);
g2 := Group([ (1,3,19,4,17)(2,13,8,11,9)(5,15,7,12,18)(6,20,14,16,10), 
              (1,5)(2,20)(3,18)(4,7)(6,13)(8,10)(9,14)(11,16)(12,19)(15,17), 
              (1,2,3,13,19,8,4,11,17,9)(5,14,15,16,7,10,12,6,18,20) ]);
g3 := Group([ (1,2,9,20,13)(3,15,5,14,18)(4,8,17,7,19)(6,11,16,12,10), 
              (1,4)(2,19)(3,6)(5,12)(7,9)(8,13)(10,15)(11,18)(14,16)(17, 20),         
              (1,3)(2,18)(4,11)(5,20)(6,8)(7,12)(9,14)(10,17)(13,15)(16,19) ]);
g4 := Group([ (2,15,12)(3,16,17)(4,10,8)(5,13,6)(9,18,20)(11,14,19), 
              (1,8)(2,4)(3,11)(5,10)(6,15)(7,20)(9,19)(12,14)(13,17)(16,18) ]);
g5 := Group([ (2,16,13)(3,10,19)(4,17,18)(5,11,9)(6,14,7)(12,15,20), 
              (1,16)(2,15)(3,14)(4,6)(5,13)(7,9)(8,18)(10,17)(11,12)(19,20) ]);

cg := CosetGeometry(g, [g1,g2,g3,g4]);

diagram := DiagramOfGeometry( cg );
DrawDiagram( diagram, "coxeter1" );

cg := CosetGeometry(g, [g1,g2,g5,g4]);

diagram := DiagramOfGeometry( cg );
DrawDiagram( diagram, "coxeter2" );
